    const { profile } = useAuth()
    const { approve, requestRevision, reject, adminOverride } = useSi()
    const requiredRole = si.chain_snapshot?.[si.current_step]?.required_role
    const optionalUser = si.chain_snapshot?.[si.current_step]?.optional_user_id
    const isAdmin = profile?.global_role === 'admin'
    // Server is the source of truth — UI gating is convenience only
    const canAct = useMemo(() => {
